```python
def with_url(label, url):
    return f"{label} ({url})" if url else label


def reg_label(source, url):
    """Classify a registration point by its actual source. `sov-print` lumps two
    very different things — archived Statements of Vote AND a figure read from a
    newspaper (NewsBank) — so distinguish them by URL."""
    u = (url or "").lower()
    if "newsbank" in u or "/news/" in u:
        return "Newspaper (via NewsBank)"
    if source == "sos-ror":
        return "CA Secretary of State — Report of Registration"
    return "Statement of Vote — participation table"


def nearest_ror(election_date):
    """The Report of Registration closest before the election, within 45 days
    (the SoS publishes a 15-day report before each statewide election). Odd-year
    interim reports fall outside the window and aren't attached to an election."""
    e = iso(election_date)
    best = None
    for r in ror:
        gap = (e - iso(r["sample"])).days
        if 0 <= gap <= 45 and (best is None or r["sample"] > best["sample"]):
            best = r
    return best
```

```python
def with_url(label, url):
    return f"{label} ({url})" if url else label


def reg_label(source, url):
    """Classify a registration point by its actual source. `sov-print` lumps two
    very different things — archived Statements of Vote AND a figure read from a
    newspaper (NewsBank) — so distinguish them by URL. (Ported verbatim from the
    retired scripts/build_franchise_csv.py.)"""
    u = (url or "").lower()
    if "newsbank" in u or "/news/" in u:
        return "Newspaper (via NewsBank)"
    if source == "sos-ror":
        return "CA Secretary of State — Report of Registration"
    return "Statement of Vote — participation table"


def load_ror(path):
    """Report-of-Registration snapshots (eligible + registered), sorted by
    sample date. Input: packages/data/registration_eligible.json."""
    ror = []
    for x in json.load(open(path)):
        ror.append({
            "sample": x["date"], "eligible": x["eligible"],
            "registered": x["registered"],
            "src": reg_label(x["source"], x.get("url", "")),
            "url": x.get("url", ""),
            "confidence": x.get("confidence", ""),
        })
    ror.sort(key=lambda r: r["sample"])
    return ror


def nearest_ror(election_date, ror):
    """The Report of Registration closest before the election, within 45 days
    (the SoS publishes a 15-day report before each statewide election). Odd-year
    interim reports fall outside the window and aren't attached to an election.
    (Ported verbatim from the retired scripts/build_franchise_csv.py, with the
    module-global `ror` made a parameter.)"""
    e = dt.date.fromisoformat(election_date)
    best = None
    for r in ror:
        gap = (e - dt.date.fromisoformat(r["sample"])).days
        if 0 <= gap <= 45 and (best is None or r["sample"] > best["sample"]):
            best = r
    return best
```

- [ ] **Step 1: Run the gate**

Run exactly:

```bash
.venv/bin/python - <<'EOF'
import csv, sys
old = {r["election_date"]: r for r in
       csv.DictReader(open("data/sf_franchise_by_election.csv"))}
new = {r["date"]: r for r in
       csv.DictReader(open("data/exports/sf_elections_consolidated.csv"))}
mismatches = []
compared = 0
for d, o in old.items():
    n = new.get(d)
    if n is None:
        mismatches.append((d, "row missing from new export"))
        continue
    for old_col, new_col in (("eligible", "eligible_voters"),
                             ("registered_ror", "registered_ror_snapshot"),
                             ("sample_date", "ror_sample_date")):
        if str(o[old_col]) != str(n[new_col]):
            mismatches.append((d, f"{old_col}: old={o[old_col]!r} new={n[new_col]!r}"))
        else:
            compared += 1
for d, why in mismatches:
    print("MISMATCH", d, why)
print(f"{'MIGRATION GATE PASS' if not mismatches else 'MIGRATION GATE FAIL'}: "
      f"{len(old)} elections, {compared} values compared, {len(mismatches)} mismatches")
sys.exit(1 if mismatches else 0)
EOF
```

Expected output (counts must match exactly this shape; the election count is the old file's row count):
`MIGRATION GATE PASS: <N> elections, <3*N> values compared, 0 mismatches`
